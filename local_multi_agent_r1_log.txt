import numpy as np
def run_regression(X, y):
    # Calculate covariance and variance
    X_mean = np.mean(X)
    varXtX = np.cov(X.T)[0, 0] * len(X)
    covXY = np.sum((X - X_mean) * (y - np.mean(y)))
    # Calculate beta and R-squared
    beta = covXY / varXtX
    m = np.mean(y)
    r2 = (covXY**2) / (m*varXtX*(len(X)-1))
    return {'beta': beta, 'r2': r2}
